Туториал для нашего квеста:

Структура:
var door1 = {Door: {Isopen: true, room1: "Спальня", room2: "Ванна", Enter: func()}	
var toothpaste = {Item: {Name: "Зубная паста", Description: "С виду обычная, но на самом деле одобренная миллионами стоматологов."}}
var toothbrush = {Item: {Name: "Зубная щетка", Description: "Новый уровень отчистки! Удаляет в два раза больше налета!"}}
var sink = {InteractiveObject: {Room: "Ванна", accessible: true, Interact: fucn()}}
var clock = {InteractiveObject: {Room: "Спальня", accessible: true, Interact: fucn()}}


{ graph: [
	Room: {
		Name: "Спальня",
		Doors: [door1],
		Slots: [],
		InteractiveItems: [clock],
	}
	Room: {
		Name: "Ванна",
		Doors: [door1],
		Slots: [Slot1: {isOpen: true, Item: toothpaste, Move: func()},
				 Slot2: {isOpen: true, Item: toothbrush, Move: func()}]
		InteractiveItems: [sink],	
	}
]}
